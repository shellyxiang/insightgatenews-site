        // 工具函数
        async function proxyGet(url) {
            const res = await fetch(`${PROXY_MULTI}/?url=${encodeURIComponent(url)}`, { cache: 'no-store' });
            if (!res.ok) throw new Error('proxy ' + res.status);
            return res;
        }

        async function fetchYahooClose(sym) {
            const url = `https://query1.finance.yahoo.com/v8/finance/chart/${sym}?range=1d&interval=1d`;
            const j = await (await proxyGet(url)).json();
            const meta = j?.chart?.result?.[0]?.meta;
            const price = meta?.regularMarketPrice ?? meta?.previousClose;
            if (!isFinite(price)) throw new Error('no price');
            const ts = new Date(meta?.regularMarketTime * 1000 || Date.now());
            
            // 转换为UTC时间
            const utcTime = ts.toISOString().slice(0, 19).replace('T', ' ');
            // 转换为北京时间 (UTC+8)
            const beijingTime = new Date(ts.getTime() + 8 * 60 * 60 * 1000).toISOString().slice(0, 19).replace('T', ' ');
            
            return { 
                value: price.toFixed(2), 
                dateLabel: `${utcTime} (UTC) / ${beijingTime} (北京时间)`
            };
        }

        // 数据获取函数
        async function fetchChinaIndex(it) {
            if (it.symbol === 'hsi') {
                return await fetchYahooClose('^HSI');
            }
            
            try {
                // 深证成指和创业板指使用.SZ后缀，其他使用.SS后缀
                const suffix = it.symbol.startsWith('399') ? '.SZ' : '.SS';
                const row = await fetchYahooClose(it.symbol + suffix);
                return row;
            } catch (e) {
                throw new Error('无法获取A股数据');
            }
        }

        async function fetchUSIndex(it) {
            try {
                const row = await fetchYahooClose(it.y);
                return row;
            } catch (e) {
                throw new Error('无法获取指数数据');
            }
        }

        async function fetchCommodity(it) {
            try {
                const row = await fetchYahooClose(it.y);
                return row;
            } catch (e) {
                throw new Error('无法获取商品数据');
            }
        }

        async function fetchFxItem(it) {
            try {
                const row = await fetchYahooClose(it.pair.replace('/', '') + '=X');
                return { ...row, value: parseFloat(row.value).toFixed(4) };
            } catch (e) {
                throw new Error('无法获取汇率数据');
            }
        }

        async function fetchCrypto(it) {
            try {
                const row = await fetchYahooClose(it.yahooSymbol);
                return row;
            } catch (e) {
                throw new Error('无法获取加密货币数据');
            }
        }

        async function loadGossip(topic) {
            const box = document.getElementById(`${topic}-gossip`);
            try {
                const res = await fetch(`${PROXY_YAHOO}/gossip?topic=${encodeURIComponent(topic)}`, {cache:'no-store'});
                const list = await res.json();

                if (!Array.isArray(list) || list.length === 0) {
                    box.innerHTML = '<div class="gossip-item">暂无数据</div>';
                    return;
                }

                box.innerHTML = list.map(x => {
                    const timeStr = x.time ? new Date(x.time).toLocaleString('zh-CN') : '—';
                    const score   = x.score || 0;
                    const source  = x.source || '—';
                    const title   = x.title || '(无标题)';
                    const url     = x.url || '#';
                    return `
                        <div class="gossip-item">
                            <a href="${url}" target="_blank" rel="noopener">${title}</a><br>
                            <small>${source} · ${timeStr} · 热度:${score}</small>
                        </div>
                    `;
                }).join('');
            } catch (e) {
                console.error('Gossip加载失败:', topic, e);
                box.innerHTML = '<div class="gossip-item">加载失败，请稍后重试</div>';
            }
        }

        // 渲染函数
        function createDataCard(it, row) {
            const card = document.createElement('div');
            card.className = 'data-card';
            
            if (row) {
                card.innerHTML = `
                    <div class="data-title">${it.name}</div>
                    <div class="data-value">${row.value}</div>
                    <div class="data-meta">来源：${row.source} · ${row.dateLabel}</div>
                `;
            } else {
                card.innerHTML = `
                    <div class="data-title">${it.name}</div>
                    <div class="data-value error">—</div>
                    <div class="data-meta">数据获取中...</div>
                `;
            }
            
            return card;
        }

        async function renderSection(items, fetcher, mountId) {
            const box = document.getElementById(mountId);
            if (!box) return;
            
            box.innerHTML = '';
            
            for (const it of items) {
                try {
                    const row = await fetcher(it);
                    const card = createDataCard(it, row);
                    box.appendChild(card);
                } catch (e) {
                    const card = createDataCard(it, null);
                    box.appendChild(card);
                }
            }
        }

            function loadCategories() {
                categoryTabs.innerHTML = '';
                financeArticles.categories.forEach(category => {
                    const tab = document.createElement('button');
                    tab.className = 'category-tab';
                    tab.textContent = category;
                    tab.dataset.category = category;
                    tab.addEventListener('click', () => loadArticles(category));

                });

                // 设置第一个标签为激活状态
                if (categoryTabs.firstChild) {
                    categoryTabs.firstChild.classList.add('active');
                }
            }

            function loadArticles(category) {
                // 更新激活的标签
                document.querySelectorAll('.category-tab').forEach(tab => {
                    tab.classList.remove('active');
                });
                document.querySelector(`[data-category="${category}"]`).classList.add('active');

                articlesContainer.innerHTML = '';

                    articlesContainer.innerHTML = `

                            <p>该分类下暂无文章</p>

                    articleElement.className = 'article-card';
                    articleElement.innerHTML = `
                        <img src="${article.image}" alt="${article.title}" onerror="this.src='https://images.unsplash.com/photo-1611974789855-9c2a0a7236a3?w=400&h=250&fit=crop'">
                        <div class="article-content">
                            <div class="article-meta">
                                <span class="article-category">${category}</span>
                                <span class="article-date">${formatDate(article.createdAt)}</span>
                            </div>
                            <h3>${article.title}</h3>
                            <p class="article-summary">${article.summary || article.content.substring(0, 150) + '...'}</p>
                            <div class="article-stats">
                                <span>👁️ ${article.views} 浏览</span>
                                <span>❤️ ${article.likes} 点赞</span>

            function setupEventListeners() {
                // 没有管理员面板，无需设置相关事件监听器

                articleFormContent.addEventListener('submit', function(e) {

            function showArticleForm(category = '') {
                articleForm.style.display = 'block';
                formTitle.textContent = '添加新文章';
                if (category) {
                    document.getElementById('article-category').value = category;
                }
                // 清除编辑状态
                delete articleForm.dataset.editCategory;
                delete articleForm.dataset.editIndex;

                articleForm.scrollIntoView({ behavior: 'smooth' });
            }

            function hideArticleForm() {
                articleForm.style.display = 'none';
                articleFormContent.reset();

            function saveArticle() {
                const category = document.getElementById('article-category').value;
                const title = document.getElementById('article-title').value;
                const content = document.getElementById('article-content').value;
                const image = document.getElementById('article-image').value || 'https://images.unsplash.com/photo-1611974789855-9c2a0a7236a3?w=400&h=250&fit=crop';
                const summary = document.getElementById('article-summary').value;
                const tags = document.getElementById('article-tags').value.split(',').map(tag => tag.trim()).filter(tag => tag);

                const articleData = {
                    id: generateId(),
                    title,
                    content,
                    image,
                    summary,
                    tags,
                    author: '管理员',
                    createdAt: new Date().toISOString(),
                    updatedAt: new Date().toISOString(),
                    views: 0,
                    likes: 0,
                    comments: []
                };

                // 检查是否是编辑模式
                if (articleForm.dataset.editCategory) {
                    // 编辑现有文章
                    const originalArticle = financeArticles.articles[articleForm.dataset.editCategory][articleForm.dataset.editIndex];
                    articleData.id = originalArticle.id;
                    articleData.createdAt = originalArticle.createdAt;
                    articleData.views = originalArticle.views;
                    articleData.likes = originalArticle.likes;
                    articleData.comments = originalArticle.comments;
                    
                    financeArticles.articles[articleForm.dataset.editCategory][articleForm.dataset.editIndex] = articleData;
                } else {
                    // 添加新文章
                    if (!financeArticles.articles[category]) {
                        financeArticles.articles[category] = [];
                    }
                    financeArticles.articles[category].push(articleData);
                }

                saveToLocalStorage();
                hideArticleForm();
                loadArticles(category);
                
                alert('文章保存成功！');
            }

            function editArticle(category, index) {
                const article = financeArticles.articles[category][index];
                document.getElementById('article-category').value = category;
                document.getElementById('article-title').value = article.title;
                document.getElementById('article-content').value = article.content;
                document.getElementById('article-image').value = article.image;
                document.getElementById('article-summary').value = article.summary || '';
                document.getElementById('article-tags').value = article.tags ? article.tags.join(', ') : '';
                
                // 保存编辑信息
                articleForm.dataset.editCategory = category;
                articleForm.dataset.editIndex = index;
                formTitle.textContent = '编辑文章';
                
                showArticleForm();
                updatePreview();
            }

            function deleteArticle(category, index) {
                if (confirm('确定要删除这篇文章吗？此操作不可撤销。')) {
                    financeArticles.articles[category].splice(index, 1);
                    saveToLocalStorage();
                    loadArticles(category);
                }
            }

            function addComment(category, index) {
                const commentInput = document.querySelector('.comment-form textarea');
                const content = commentInput.value.trim();
                
                if (!content) {
                    alert('请输入评论内容');
                    return;
                }

                const comment = {
                    id: generateId(),
                    author: '访客', // 实际应用中应该从用户系统获取
                    content: content,
                    createdAt: new Date().toISOString()
                };

                if (!financeArticles.articles[category][index].comments) {
                    financeArticles.articles[category][index].comments = [];
                }
                financeArticles.articles[category][index].comments.push(comment);
                saveToLocalStorage();
                
                // 刷新评论显示
                viewArticleDetail(category, index);
                commentInput.value = '';
            }

            // 工具函数
            function generateId() {
                return 'id-' + Date.now() + '-' + Math.random().toString(36).substr(2, 9);
            }

            function formatDate(dateString) {
                const date = new Date(dateString);
                return date.toLocaleDateString('zh-CN') + ' ' + date.toLocaleTimeString('zh-CN', {hour: '2-digit', minute: '2-digit'});
            }

            function saveToLocalStorage() {
                localStorage.setItem('financeArticles', JSON.stringify(financeArticles));
            }

            // 富文本编辑功能
            function formatText(command) {
                const textarea = document.getElementById('article-content');
                const start = textarea.selectionStart;
                const end = textarea.selectionEnd;
                const selectedText = textarea.value.substring(start, end);
                
                let formattedText = '';
                switch(command) {
                    case 'bold':
                        formattedText = `**${selectedText}**`;
                        break;
                    case 'italic':
                        formattedText = `*${selectedText}*`;
                        break;
                    case 'underline':
                        formattedText = `__${selectedText}__`;
                        break;
                }
                
                textarea.value = textarea.value.substring(0, start) + formattedText + textarea.value.substring(end);
                textarea.focus();
                textarea.setSelectionRange(start + formattedText.length, start + formattedText.length);
                updatePreview();
            }

            function insertImage() {
                const url = prompt('请输入图片URL：');
                if (url) {
                    const textarea = document.getElementById('article-content');
                    const cursorPos = textarea.selectionStart;
                    const imageMarkdown = `\n![图片描述](${url})\n`;
                    textarea.value = textarea.value.substring(0, cursorPos) + imageMarkdown + textarea.value.substring(cursorPos);
                    updatePreview();
                }
            }

            function insertLink() {
                const url = prompt('请输入链接URL：');
                const text = prompt('请输入链接文本：');
                if (url && text) {
                    const textarea = document.getElementById('article-content');
                    const cursorPos = textarea.selectionStart;
                    const linkMarkdown = `[${text}](${url})`;
                    textarea.value = textarea.value.substring(0, cursorPos) + linkMarkdown + textarea.value.substring(cursorPos);
                    updatePreview();
                }
            }

            function insertList(type) {
                const textarea = document.getElementById('article-content');
                const cursorPos = textarea.selectionStart;
                const listItem = type === 'ul' ? '- 列表项\n' : '1. 列表项\n';
                textarea.value = textarea.value.substring(0, cursorPos) + listItem + textarea.value.substring(cursorPos);
                updatePreview();
            }

            function insertHeading() {
                const textarea = document.getElementById('article-content');
                const cursorPos = textarea.selectionStart;
                const heading = '## 标题\n';
                textarea.value = textarea.value.substring(0, cursorPos) + heading + textarea.value.substring(cursorPos);
                updatePreview();
            }

            // 更新预览
            function updatePreview() {
                const content = document.getElementById('article-content').value;
                const preview = document.getElementById('content-preview');
                preview.innerHTML = markdownToHtml(content);
            }

            // 简单的Markdown转HTML
            function markdownToHtml(text) {
                return text
                    .replace(/##\s*(.*?)\n/g, '<h2>$1</h2>')
                    .replace(/###\s*(.*?)\n/g, '<h3>$1</h3>')
                    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
                    .replace(/\*(.*?)\*/g, '<em>$1</em>')
                    .replace(/__(.*?)__/g, '<u>$1</u>')
                    .replace(/\r\n|\n/g, '<br>')
                    .replace(/!\[(.*?)]\((.*?)\)/g, '<img src="$2" alt="$1" style="max-width: 100%; height: auto; margin: 10px 0;">')
                    .replace(/\[(.*?)]\((.*?)\)/g, '<a href="$2" target="_blank">$1</a>')
                    .replace(/-\s*(.*?)(<br>|$)/g, '<li>$1</li>')
                    .replace(/\d+\.\s*(.*?)(<br>|$)/g, '<li>$1</li>');
            }

            // 预览文章
            function previewArticle() {
                const title = document.getElementById('article-title').value;
                const content = document.getElementById('article-content').value;
                const image = document.getElementById('article-image').value;
                const summary = document.getElementById('article-summary').value;
                
                let previewHtml = `
                    <div class="article-preview">
                        <h2>${title || '文章标题'}</h2>
                        ${image ? `<img src="${image}" alt="${title}" style="max-width: 100%; height: auto; margin: 20px 0;">` : ''}
                        ${summary ? `<p class="preview-summary"><strong>摘要：</strong>${summary}</p>` : ''}
                        <div class="preview-content">${markdownToHtml(content)}</div>
                    </div>
                `;
                
                document.getElementById('preview-modal-content').innerHTML = previewHtml;
                document.getElementById('article-preview-modal').style.display = 'block';
            }

            function closePreview() {
                document.getElementById('article-preview-modal').style.display = 'none';
            }

        /* 金币炸开动画功能 */
        function createCoinExplosion(x, y) {
            const explosion = document.createElement('div');
            explosion.className = 'coin-explosion';
            explosion.style.left = x + 'px';
            explosion.style.top = y + 'px';
            
            // 创建金币粒子（20-30个）
            const coinCount = Math.floor(Math.random() * 10) + 20;
            
            for (let i = 0; i < coinCount; i++) {
                const coin = createCoin();
                explosion.appendChild(coin);
                
                // 创建轨迹效果
                if (Math.random() > 0.7) {
                    const trail = createCoinTrail();
                    explosion.appendChild(trail);
                }
            }
            
            document.body.appendChild(explosion);
            
            // 动画结束后移除元素
            setTimeout(() => {
                explosion.remove();
            }, 1200);
        }
        
        function createCoin() {
            const coin = document.createElement('div');
            coin.className = 'coin';
            
            // 随机角度和距离
            const angle = Math.random() * Math.PI * 2;
            const distance = Math.random() * 150 + 50;
            const tx = Math.cos(angle) * distance;
            const ty = Math.sin(angle) * distance;
            const rotation = Math.random() * 720 + 180;
            
            // 设置CSS变量
            coin.style.setProperty('--tx', tx + 'px');
            coin.style.setProperty('--ty', ty + 'px');
            coin.style.setProperty('--r', rotation + 'deg');
            
            // 随机延迟和持续时间
            const delay = Math.random() * 0.3;
            const duration = 0.8 + Math.random() * 0.4;
            coin.style.animationDelay = delay + 's';
            coin.style.animationDuration = duration + 's';
            
            // 随机大小变化
            const size = 10 + Math.random() * 8;
            coin.style.width = size + 'px';
            coin.style.height = size + 'px';
            
            // 创建光泽效果
            const shine = document.createElement('div');
            shine.className = 'coin-shine';
            coin.appendChild(shine);
            
            return coin;
        }
        
        function createCoinTrail() {
            const trail = document.createElement('div');
            trail.className = 'coin-trail';
            
            // 随机位置和大小
            const size = 1 + Math.random() * 2;
            trail.style.width = size + 'px';
            trail.style.height = size + 'px';
            trail.style.left = (Math.random() * 40 - 20) + 'px';
            trail.style.top = (Math.random() * 40 - 20) + 'px';
            
            // 随机延迟和持续时间
            const delay = Math.random() * 0.5;
            const duration = 0.3 + Math.random() * 0.3;
            trail.style.animationDelay = delay + 's';
            trail.style.animationDuration = duration + 's';
            
            return trail;
        }
        
        // 为导航链接添加点击效果
        document.addEventListener('DOMContentLoaded', function() {
            const navLinks = document.querySelectorAll('.nav-link');
            
            navLinks.forEach(link => {
                link.addEventListener('click', function(e) {
                    const rect = this.getBoundingClientRect();
                    const x = rect.left + rect.width / 2;
                    const y = rect.top + rect.height / 2;
                    
                    // 创建金币炸开效果
                    createCoinExplosion(x, y);
                    
                    // 添加点击反馈效果
                    this.style.transform = 'scale(0.95)';
                    setTimeout(() => {
                        this.style.transform = 'scale(1)';
                    }, 150);
                });
            });
        });
