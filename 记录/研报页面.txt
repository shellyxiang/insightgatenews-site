    <script>
        // 每日研报页面功能
        document.addEventListener('DOMContentLoaded', function() {
            const uploadBox = document.getElementById('upload-box');
            const pdfUpload = document.getElementById('pdf-upload');
            const uploadProgress = document.getElementById('upload-progress');
            const progressFill = document.getElementById('progress-fill');

            // 拖拽上传功能
            uploadBox.addEventListener('dragover', function(e) {
                e.preventDefault();
                uploadBox.style.borderColor = 'var(--accent-green)';
                uploadBox.style.backgroundColor = 'rgba(74, 124, 89, 0.1)';
            });

            uploadBox.addEventListener('dragleave', function(e) {
                e.preventDefault();
                uploadBox.style.borderColor = 'var(--border-color)';
                uploadBox.style.backgroundColor = 'var(--background-light)';
            });

            uploadBox.addEventListener('drop', function(e) {
                e.preventDefault();
                uploadBox.style.borderColor = 'var(--border-color)';
                uploadBox.style.backgroundColor = 'var(--background-light)';
                
                const files = e.dataTransfer.files;
                if (files.length > 0) {
                    handleFileUpload(files[0]);
                }
            });

            // 文件选择上传
            pdfUpload.addEventListener('change', function(e) {
                if (e.target.files.length > 0) {
                    handleFileUpload(e.target.files[0]);
                }
            });


            async function handleFileUpload(file) {
                if (file.type !== 'application/pdf') {
                    alert('请上传PDF格式的文件');
                    return;
                }

                if (file.size > 50 * 1024 * 1024) {
                    alert('文件大小不能超过50MB');
                    return;
                }

                // 显示上传进度条（简单版本，真实进度以后再做）
                uploadProgress.style.display = 'block';
                progressFill.style.width = '30%';

                // 用 FormData 装 PDF 文件
                const formData = new FormData();
                formData.append('file', file);

                try {
                    // 发送 POST 请求到你的 Node.js 后端
                    const response = await fetch('http://localhost:3000/api/upload-report', {
                        method: 'POST',
                        body: formData
                    });

                    progressFill.style.width = '80%';

                    if (!response.ok) {
                        throw new Error('上传失败');
                    }

                    const result = await response.json();

                    progressFill.style.width = '100%';
                    setTimeout(() => {
                        uploadProgress.style.display = 'none';
                    }, 500);

                    alert('文件上传成功！');

                    // 显示到页面上
                    addNewReport(result.file.savedName);

                } catch (err) {
                    console.error(err);
                    alert('上传失败，请检查服务器是否已启动');
                    uploadProgress.style.display = 'none';
                }
            }



            // 研报数据
            // 从后端返回的真实数据
            let reportsData = [];

            // 把字节大小转成 MB 文本
            function formatSize(size) {
                if (!size && size !== 0) return '';
                const mb = size / (1024 * 1024);
                return mb.toFixed(2) + ' MB';
            }

            // 从文件名里提取日期（你是 _YYYYMMDD 结尾）
            function extractDateFromFilename(filename) {
                const noExt = filename.replace(/\.pdf$/i, '');
                const parts = noExt.split('_');
                const last = parts[parts.length - 1]; // 20251121
                if (last && last.length === 8 && /^\d+$/.test(last)) {
                    const y = last.slice(0, 4);
                    const m = last.slice(4, 6);
                    const d = last.slice(6, 8);
                    return `${y}-${m}-${d}`;
                }
                return '';
            }

            // 调后端 API，获取 uploads 里的文件列表
            async function loadReportsFromServer() {
                try {
                    const res = await fetch('http://localhost:3000/api/reports');
                    if (!res.ok) {
                        throw new Error('获取列表失败');
                    }
                    const files = await res.json();

                    // 把后端返回的文件信息变成前端用的结构
                    reportsData = files.map(file => {
                        const filename = file.filename;
                        const title = filename.replace(/\.pdf$/i, '');
                        const date = extractDateFromFilename(filename);
                        return {
                            filename,
                            title,
                            date,
                            fileSize: formatSize(file.size),
                            category: '用户上传',
                            summary: '上传的研究报告'
                        };
                    });

                    loadReports();
                } catch (err) {
                    console.error(err);
                    alert('无法加载研报列表，请检查服务器是否在运行');
                }
            }





            const reportsContainer = document.getElementById('reports-container');

            function loadReports() {
                reportsContainer.innerHTML = '';

                reportsData.forEach(report => {
                    const reportElement = document.createElement('div');
                    reportElement.className = 'report-card';

                    // 下载链接：指向我们的静态目录
                    const downloadUrl = `http://localhost:3000/uploads/${encodeURIComponent(report.filename)}`;

                    reportElement.innerHTML = `
                        <div class="report-header">
                            <span class="report-category">${report.category}</span>
                            <span class="report-date">${report.date || ''}</span>
                        </div>
                        <h3>${report.title}</h3>
                        <p class="report-summary">${report.summary}</p>
                        <div class="report-meta">
                            <span class="report-author">文件名：${report.filename}</span>
                            <span class="report-info">${report.fileSize}</span>
                        </div>
                        <div class="report-actions">
                            <button class="btn-download">下载PDF</button>
                            <button class="btn-preview">在线预览</button>
                        </div>
                    `;

                    // 绑定下载和预览按钮
                    const downloadBtn = reportElement.querySelector('.btn-download');
                    const previewBtn = reportElement.querySelector('.btn-preview');

                    downloadBtn.addEventListener('click', () => {
                        window.open(downloadUrl, '_blank');
                    });

                    // 预览先简单用浏览器内置 PDF 预览
                    previewBtn.addEventListener('click', () => {
                        window.open(downloadUrl, '_blank');
                    });

                    reportsContainer.appendChild(reportElement);
                });
            }


            function addNewReport(filename) {
                const newReport = {
                    title: filename.replace('.pdf', ''),
                    author: '用户上传',
                    date: new Date().toISOString().split('T')[0],
                    summary: '用户上传的研究报告',
                    pages: '未知',
                    fileSize: '待计算',
                    category: '用户上传'
                };
                reportsData.unshift(newReport);
                loadReports();
            }

            loadReportsFromServer();
        });
    </script>
